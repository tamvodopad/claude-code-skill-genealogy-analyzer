```python
def orthodox_easter_julian(year: int) -> tuple:
    """
    –†–∞—Å—á—ë—Ç –ø—Ä–∞–≤–æ—Å–ª–∞–≤–Ω–æ–π –ü–∞—Å—Ö–∏ –ø–æ —é–ª–∏–∞–Ω—Å–∫–æ–º—É –∫–∞–ª–µ–Ω–¥–∞—Ä—é.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç (–º–µ—Å—è—Ü, –¥–µ–Ω—å) –ø–æ —Å—Ç–∞—Ä–æ–º—É —Å—Ç–∏–ª—é.
    """
    a = year % 4
    b = year % 7
    c = year % 19
    d = (19 * c + 15) % 30
    e = (2 * a + 4 * b - d + 34) % 7
    month = (d + e + 114) // 31
    day = ((d + e + 114) % 31) + 1
    return (month, day)

def julian_to_gregorian(year: int, month: int, day: int) -> tuple:
    """
    –ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è —é–ª–∏–∞–Ω—Å–∫–æ–π –¥–∞—Ç—ã –≤ –≥—Ä–∏–≥–æ—Ä–∏–∞–Ω—Å–∫—É—é.
    –î–æ–±–∞–≤–ª—è–µ—Ç 12 –¥–Ω–µ–π –¥–ª—è 19 –≤–µ–∫–∞, 13 –¥–Ω–µ–π –¥–ª—è 20 –≤–µ–∫–∞.
    """
    from datetime import date, timedelta

    if year < 1900:
        delta = 12
    else:
        delta = 13

    julian = date(year, month, day)
    gregorian = julian + timedelta(days=delta)
    return (gregorian.year, gregorian.month, gregorian.day)
```
